#include <stdio.h>
#include <string.h>

#define MAX 50
#define EPS 2

int q[MAX][3];
int state = 1;

void symbol(char ch) {
    if (ch == 'a')
        q[state][0] = state + 1;
    else if (ch == 'b')
        q[state][1] = state + 1;
    state++;
}

void union_ab(char a, char b) {
    int s = state;
    q[s][EPS] = (s + 1) * 10 + (s + 3);
    state++;
    symbol(a);
    q[state][EPS] = s + 5;
    state++;
    symbol(b);
    q[state][EPS] = s + 5;
    state++;
}

void star(char a) {
    int s = state;
    q[s][EPS] = (s + 1) * 10 + (s + 3);
    state++;
    symbol(a);
    q[state][EPS] = (s + 1) * 10 + (state + 1);
    state++;
}

int main() {
    char reg[30];
    printf("Enter regular expression: ");
    scanf("%s", reg);

    for (int i = 0; i < strlen(reg); i++) {
        if (reg[i] == 'a' || reg[i] == 'b') {
            if (reg[i + 1] == '*') {
                star(reg[i]);
                i++;
            }
            else if (reg[i + 1] == '|') {
                union_ab(reg[i], reg[i + 2]);
                i += 2;
            }
            else {
                symbol(reg[i]);
            }
        }
    }

    printf("\nTransition Table:\n");
    for (int i = 1; i < state; i++) {
        if (q[i][0])
            printf("q[%d,a] -> %d\n", i, q[i][0]);
        if (q[i][1])
            printf("q[%d,b] -> %d\n", i, q[i][1]);
    }

    return 0;
}

#include <stdio.h>
#include <string.h>
#define MAX 50
#define EPS 2

int epsilon[MAX][MAX];
n;int visited[MAX];
int 

void dfs(int s) {
    visited[s] = 1;
    for (int i = 1; i <= n; i++) {
        if (epsilon[s][i] == 1 && !visited[i])
            dfs(i);
    }
}

void build_epsilon_matrix() {
    int i, j;
    for (i = 0; i < MAX; i++)
        for (j = 0; j < MAX; j++)
            epsilon[i][j] = 0;
}

int main() {
    char reg[30];
    printf("Enter regular expression: ");
    scanf("%s", reg);

    n = 10;
    build_epsilon_matrix();

    printf("\nTransition Matrix:\n");
    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= n; j++)
            printf("%d ", epsilon[i][j]);
        printf("\n");
    }

    printf("\nEpsilon Closures:\n");
    for (int i = 1; i <= n; i++) {
        for (int j = 0; j < MAX; j++)
            visited[j] = 0;
        dfs(i);
        printf("ε-closure(%d) = { ", i);
        for (int j = 1; j <= n; j++)
            if (visited[j])
                printf("%d ", j);
        printf("}\n");
    }

    return 0;
}

#include <stdio.h>
#include <string.h>
#include <ctype.h>

char p[20][20], f[26][20];
int c[26], n;

void add(char A, char x)
{
    int i, k = A - 'A';
    for(i=0;i<c[k];i++)
        if(f[k][i]==x) return;
    f[k][c[k]++] = x;
}

void first(char A)
{
    int i,j,k;

    for(i=0;i<n;i++)
    {
        if(p[i][0]==A)
        {
            for(j=3;p[i][j]!='\0';j++)
            {
                if(p[i][j]=='|') continue;

                if(!isupper(p[i][j]))
                    add(A,p[i][j]);
                else
                {
                    first(p[i][j]);
                    for(k=0;k<c[p[i][j]-'A'];k++)
                        add(A,f[p[i][j]-'A'][k]);
                }
                break;
            }
        }
    }
}

int main()
{
    int i,j;

    printf("Enter number of productions: ");
    scanf("%d",&n);

    printf("Enter productions:\n");
    for(i=0;i<n;i++)
        scanf("%s",p[i]);

    for(i=0;i<n;i++)
        first(p[i][0]);

    printf("\nFIRST sets:\n");

    for(i=0;i<26;i++)
        if(c[i])
        {
            printf("FIRST(%c) = { ",i+'A');
            for(j=0;j<c[i];j++)
                printf("%c ",f[i][j]);

        }
}

#include <stdio.h>
#include <string.h>
#define MAX 50
int main()
{
    int n, i;
    char line[MAX];
    char lhs[10], rhs[20];
    char arg1[10], arg2[10], op;
    printf("Enter number of statements: ");
    scanf("%d", &n);
    getchar();
    printf("Enter Three Address Code:\n");
    for(i = 0; i < n; i++)
    {
        fgets(line, sizeof(line), stdin);
        line[strcspn(line, "\n")] = '\0';
        sscanf(line, "%[^=]=%s", lhs, rhs);
        if (sscanf(rhs, "%[^+-*/]%c%s", arg1, &op, arg2) == 3)
        {
            printf("\nLDA R1, %s\n", arg1);
            printf("LDA R2, %s\n", arg2);

            if(op == '+') printf("ADD R2, R1\n");
            else if(op == '-') printf("SUB R2, R1\n");
            else if(op == '*') printf("MUL R2, R1\n");
            else if(op == '/') printf("DIV R2, R1\n");
            printf("STA %s, R2\n", lhs);
        }
        else
        {
            printf("\nLDA R1, %s\n", rhs);
            printf("STA %s, R1\n", lhs);
        }
    }
    return 0;
}
